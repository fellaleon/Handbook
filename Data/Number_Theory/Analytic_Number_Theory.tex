\ee
\end{theorem}

\subsection{Gcd-sum function}

\begin{definition}[gcd-sum function]
\be
g(n) = \sum^n_{k=1}\gcd(k,n).
\ee
\end{definition}

\begin{theorem}
For every prime number $p$ and positive integer $\alpha\geq 1$,
\be
g\brb{p^{\alpha}} = \brb{\alpha+1}p^{\alpha} - \alpha p^{\alpha -1}.
\ee
\end{theorem}
